import matplotlib.pyplot as plt
import pandas as pd

from techminer import DataFrame, Plot, heatmap, extract_country

#
# Data loading
#
df = DataFrame(
    pd.read_json(
        "https://raw.githubusercontent.com/jdvelasq/techminer/master/data/tutorial/"
        + "cleaned-data.json",
        orient="records",
        lines=True,
    )
)
